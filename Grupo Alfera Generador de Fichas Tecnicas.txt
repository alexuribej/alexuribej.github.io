async function generate() {
  const inputs = document.querySelectorAll('.prod-link');
  const urls = Array.from(inputs).map(i => i.value.trim()).filter(v => v !== "");
  
  if (urls.length === 0) { alert("Por favor ingresa al menos un link."); return; }

  const btn = document.getElementById("genBtn");
  const loading = document.getElementById("loading");
  const result = document.getElementById("result");

  btn.disabled = true;
  loading.style.display = "block";
  result.style.display = "none";

  const prompt = `Actúa como un experto en licitaciones de Mercado Público Chile. 
  Genera una ficha técnica profesional en formato JSON para estos productos: ${urls.join(", ")}. 
  Para cada producto incluye: nombre, descripción_breve, especificaciones (objeto con 6-8 características), beneficios y aplicaciones.`;

  try {
    const res = await fetch("https://openrouter.ai/api/v1/chat/completions", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": "Bearer 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",
        "HTTP-Referer": "https://grupoalfera.cl",
        "X-Title": "Ficha Tecnica Alfera"
      },
      body: JSON.stringify({
        model: "meta-llama/llama-3.3-70b-instruct",
        messages: [{ role: "user", content: prompt }]
      })
    });

    const data = await response.json();
    const content = data.choices[0].message.content;

    // Crear el Blob para descargar
    const blob = new Blob([content], { type: 'text/html' });
    const url = URL.createObjectURL(blob);
    const dl = document.getElementById("downloadLink");
    dl.href = url;
    dl.download = "Ficha_Tecnica_Alfera.html";

    result.style.display = "block";
  } catch (e) {
    alert("Error de conexión. Revisa el saldo de OpenRouter.");
  } finally {
    btn.disabled = false;
    loading.style.display = "none";
  }
}
